# 中1テストキット V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a PWA for Japanese junior high 1st-year midterm test prep covering 5 subjects, with one-time 980円 purchase, drill + AI question + mock exam features, deployed on Cloudflare.

**Architecture:** Single Next.js 16 (App Router) project deployed to Cloudflare Pages via @opennextjs/cloudflare, using D1 (SQLite) for data, KV for rate limiting, R2 for image uploads. API routes live inside the Next.js app (no separate Hono service — simpler than spec, achieves same goals). Magic-link auth via Resend, Stripe Checkout for payment, Anthropic Claude Haiku 4.5 for AI tutoring (Sonnet 4.6 for offline content generation).

**Tech Stack:** Next.js 16, TypeScript, Tailwind CSS, Vitest, Drizzle ORM (D1), Stripe, Resend, Anthropic SDK, @opennextjs/cloudflare, Cloudflare Pages/Workers/D1/KV/R2.

**Spec:** [docs/superpowers/specs/2026-05-08-中1テストキット-design.md](../specs/2026-05-08-中1テストキット-design.md)

```ts
import { SignJWT, jwtVerify } from "jose";

const ALG = "HS256";

export async function signSessionToken(opts: {
  userId: string;
  secret: string;
  expiresIn?: string;
}): Promise<string> {
  const secret = new TextEncoder().encode(opts.secret);
  return new SignJWT({ userId: opts.userId })
    .setProtectedHeader({ alg: ALG })
    .setIssuedAt()
    .setExpirationTime(opts.expiresIn ?? "30d")
    .sign(secret);
}

export async function verifySessionToken(opts: {
  token: string;
  secret: string;
}): Promise<{ userId: string }> {
  const secret = new TextEncoder().encode(opts.secret);
  const { payload } = await jwtVerify(opts.token, secret);
  if (typeof payload.userId !== "string") {
    throw new Error("invalid payload");
  }
  return { userId: payload.userId };
}
```

```ts
import { SignJWT, jwtVerify } from "jose";

const ALG = "HS256";

export async function signMagicLinkToken(opts: {
  email: string;
  secret: string;
}): Promise<string> {
  const secret = new TextEncoder().encode(opts.secret);
  return new SignJWT({ email: opts.email, purpose: "magic-link" })
    .setProtectedHeader({ alg: ALG })
    .setIssuedAt()
    .setExpirationTime("15m")
    .sign(secret);
}

export async function verifyMagicLinkToken(opts: {
  token: string;
  secret: string;
}): Promise<{ email: string }> {
  const secret = new TextEncoder().encode(opts.secret);
  const { payload } = await jwtVerify(opts.token, secret);
  if (payload.purpose !== "magic-link" || typeof payload.email !== "string") {
    throw new Error("invalid payload");
  }
  return { email: payload.email };
}
```

```ts
export async function getCurrentUser() {
  const secret = process.env.JWT_SECRET;
  if (!secret) return null;
  const cookieStore = await cookies();
  const session = cookieStore.get("session")?.value;
  if (!session) return null;
  try {
    const { userId } = await verifySessionToken({ token: session, secret });
    const db = getDb();
    return await db.select().from(users).where(eq(users.id, userId)).get();
  } catch {

  }
}

export async function requireUser() {
  const user = await getCurrentUser();
  if (!user) throw new Response("unauthorized", { status: 401 });
  return user;

export async function requireActiveUser() {
  const user = await requireUser();
  if (!user.expiresAt || user.expiresAt.getTime() < Date.now()) {
    throw new Response("expired", { status: 402 });

  return user;
```

```ts
import { getStripe } from "@/lib/stripe";
import { getDb } from "@/db/client";
import { users } from "@/db/schema";
import { eq } from "drizzle-orm";
import { nanoid } from "nanoid";
import { signMagicLinkToken } from "@/lib/magic-link";
import { sendMagicLinkEmail } from "@/lib/email";

const ACCESS_DAYS = 30;

export async function POST(req: Request) {
  const stripe = getStripe();
  const sig = req.headers.get("stripe-signature");
  const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET;
  if (!sig || !webhookSecret) {
    return new Response("missing signature/secret", { status: 400 });
  }

  const body = await req.text();
  let event: import("stripe").Stripe.Event;
  try {
    event = await stripe.webhooks.constructEventAsync(body, sig, webhookSecret);
  } catch (err) {
    return new Response(`invalid signature: ${(err as Error).message}`, { status: 400 });
  }

  if (event.type !== "checkout.session.completed") {
    return Response.json({ received: true });
  }

  const session = event.data.object;
  const email = session.customer_details?.email ?? session.metadata?.email;
  if (!email) return new Response("no email on session", { status: 400 });

  const db = getDb();
  const now = new Date();
  const expires = new Date(now.getTime() + ACCESS_DAYS * 24 * 3600 * 1000);

  const existing = await db.select().from(users).where(eq(users.email, email)).get();
  let userId: string;
  if (existing) {
    userId = existing.id;
    await db
      .update(users)
      .set({ purchasedAt: now, expiresAt: expires, updatedAt: now })
      .where(eq(users.id, userId));
  } else {
    userId = nanoid();
    await db.insert(users).values({
      id: userId,
      email,
      purchasedAt: now,
      expiresAt: expires,
    });
  }

  const jwtSecret = process.env.JWT_SECRET!;
  const token = await signMagicLinkToken({ email, secret: jwtSecret });
  const appUrl = process.env.APP_URL ?? "http://localhost:3000";
  await sendMagicLinkEmail({
    apiKey: process.env.RESEND_API_KEY!,
    fromEmail: process.env.RESEND_FROM_EMAIL!,
    toEmail: email,
    link: `${appUrl}/api/auth/verify?token=${encodeURIComponent(token)}`,
  });

  return Response.json({ received: true });
}
```

```ts
import fs from "node:fs";
import path from "node:path";
import { execFileSync } from "node:child_process";
import type { ContentTopic } from "../src/content/types";
import { sugakuPositiveNegative } from "../src/content/seed/sugaku-positive-negative";

function escapeSQL(s: string): string {
  return s.replace(/'/g, "''");
}

function loadAll(): ContentTopic[] {
  const result: ContentTopic[] = [sugakuPositiveNegative];
  const dir = path.resolve("src/content/generated");
  if (fs.existsSync(dir)) {
    for (const file of fs.readdirSync(dir)) {
      if (!file.endsWith(".json")) continue;
      const text = fs.readFileSync(path.join(dir, file), "utf8");
      result.push(JSON.parse(text));
    }
  }
  return result;
}

function buildSQL(topics: ContentTopic[]): string {
  const stmts: string[] = [];
  let topicId = 100;
  let questionId = 1000;
  let order = 1;
  for (const t of topics) {
    stmts.push(
      `INSERT INTO topics (id, subject_id, slug, name, "order", curriculum_ref) VALUES (${topicId}, (SELECT id FROM subjects WHERE slug='${t.subjectSlug}'), '${t.topicSlug}', '${escapeSQL(t.topicName)}', ${order}, ${t.curriculumRef ? `'${escapeSQL(t.curriculumRef)}'` : "NULL"});`
    );
    stmts.push(
      `INSERT INTO lessons (topic_id, content_md, estimated_minutes) VALUES (${topicId}, '${escapeSQL(t.lesson.contentMd)}', ${t.lesson.estimatedMinutes});`
    );
    for (const q of t.questions) {
      stmts.push(
        `INSERT INTO questions (id, topic_id, type, question_text, choices, answer, explanation, difficulty) VALUES (${questionId}, ${topicId}, '${q.type}', '${escapeSQL(q.questionText)}', ${q.choices ? `'${escapeSQL(JSON.stringify(q.choices))}'` : "NULL"}, '${escapeSQL(q.answer)}', '${escapeSQL(q.explanation)}', ${q.difficulty});`
      );
      questionId += 1;
    }
    topicId += 1;
    order += 1;
  }
  return stmts.join("\n");
}

const sql = buildSQL(loadAll());
const tmpFile = path.resolve(".tmp-seed.sql");
fs.writeFileSync(tmpFile, sql);
```

```ts
const MASTERY_THRESHOLD = 5;

export function computeNextMastery(input: {
  current: number;
  consecutive: number;
  isCorrect: boolean;
}): { masteryLevel: number; consecutiveCorrect: number } {
  if (!input.isCorrect) {
    return { masteryLevel: input.current, consecutiveCorrect: 0 };
  }
  const consecutive = input.consecutive + 1;
  let level = input.current;
  if (level === 0) level = 1;
  if (consecutive >= MASTERY_THRESHOLD && level < 2) level = 2;
  return { masteryLevel: level, consecutiveCorrect: consecutive };
}
```

```ts
const DAILY_LIMIT = 30;

function todayKey(userId: string): string {
  const date = new Date().toISOString().slice(0, 10);
  return `ai:${userId}:${date}`;
}

export async function checkAndIncrement(userId: string): Promise<{
  allowed: boolean;
  remaining: number;
}> {
  const { env } = getCloudflareContext();
  const key = todayKey(userId);
  const raw = await env.AI_RATE_LIMIT.get(key);
  const used = raw ? parseInt(raw, 10) : 0;
  if (used >= DAILY_LIMIT) {
    return { allowed: false, remaining: 0 };
  }
  await env.AI_RATE_LIMIT.put(key, String(used + 1), { expirationTtl: 60 * 60 * 26 });
  return { allowed: true, remaining: DAILY_LIMIT - (used + 1) };
}
```

```ts
export async function streamAnswer(prompt: string): Promise<ReadableStream<Uint8Array>> {

    max_tokens: 1024,

    messages: [{ role: "user", content: prompt }],

  return new ReadableStream({

      for await (const event of stream) {

          controller.enqueue(encoder.encode(event.delta.text));
        }
      }

    },
```

```ts
});

export async function POST(req: Request) {

  const json = await req.json().catch(() => null);
  const parsed = Body.safeParse(json);
  if (!parsed.success) {
    return Response.json({ error: "invalid body" }, { status: 400 });

      { error: "daily limit reached, try tomorrow" },

      userId: user.id,
      topicId: parsed.data.topicId ?? null,
      prompt: parsed.data.prompt,
      response: captured,
    });
  })();

  return new Response(tee[0], {
    headers: { "Content-Type": "text/plain; charset=utf-8" },
  });
```

```ts
    totalTopics: number;
    masteredTopics: number;
    learningTopics: number;
  }> = [];

  for (const s of subjectRows) {
    const allTopics = await db.select().from(topics).where(eq(topics.subjectId, s.id));
    const progressRows = await db
      .select()
      .from(topicProgress)
      .where(eq(topicProgress.userId, user.id));
    const progressByTopic = new Map(progressRows.map((r) => [r.topicId, r.masteryLevel]));
    let mastered = 0;
    let learning = 0;
    for (const t of allTopics) {
      const lvl = progressByTopic.get(t.id) ?? 0;
      if (lvl >= 2) mastered += 1;
      else if (lvl === 1) learning += 1;
    }
    result.push({
      subject: { id: s.id, slug: s.slug, name: s.name },
      totalTopics: allTopics.length,
      masteredTopics: mastered,
      learningTopics: learning,
    });
  }

  const attemptCount = await db
    .select({ count: sql<number>`count(*)` })
    .from(attempts)
    .where(eq(attempts.userId, user.id))
    .get();

  return Response.json({
    bySubject: result,
    totalAttempts: attemptCount?.count ?? 0,
  });
}
```

```tsx
"use client";
import { useEffect, useState } from "react";
import Link from "next/link";

type Question = {
  id: number;
  type: "multi_choice" | "short_answer" | "numeric";
  questionText: string;
  choices: string[] | null;
};

type AttemptResponse = {
  isCorrect: boolean;
  correctAnswer: string;
  explanation: string;
  masteryLevel: number;
};

export function DrillRunner({ topicId }: { topicId: number }) {
  const [questions, setQuestions] = useState<Question[]>([]);
  const [idx, setIdx] = useState(0);
  const [answer, setAnswer] = useState("");
  const [feedback, setFeedback] = useState<AttemptResponse | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch(`/api/topics/${topicId}/questions`)
      .then((r) => r.json())
      .then((data) => {
        setQuestions(data.questions);
        setLoading(false);
      });
  }, [topicId]);

  if (loading) return <p>読み込み中...</p>;
  if (questions.length === 0) return <p>問題がありません</p>;
  const q = questions[idx];
  if (!q) {
    return (
      <div className="space-y-4">
        <h2 className="text-xl font-bold">完了！</h2>
        <Link href={`/topics/${topicId}`} className="block bg-stone-900 text-white text-center py-3 rounded-lg">
          トピックへ戻る
        </Link>
      </div>
    );
  }

  async function submit() {
    const res = await fetch("/api/attempts", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ questionId: q!.id, userAnswer: answer }),
    });
    setFeedback(await res.json());
  }

  function next() {
    setFeedback(null);
    setAnswer("");
    setIdx((i) => i + 1);
  }

  return (
    <div className="space-y-4">
      <p className="text-sm text-stone-500">問題 {idx + 1} / {questions.length}</p>
      <p className="text-lg font-medium">{q.questionText}</p>
      {q.type === "multi_choice" && q.choices ? (
        <div className="space-y-2">
          {q.choices.map((c) => (
            <button
              key={c}
              type="button"
              disabled={!!feedback}
              onClick={() => setAnswer(c)}
              className={`w-full text-left border rounded-lg p-3 ${
                answer === c ? "border-emerald-600 bg-emerald-50" : "border-stone-300"
              }`}
            >
              {c}
            </button>
          ))}
        </div>
      ) : (
        <input
          type="text"
```

```tsx
"use client";
import { useState } from "react";

export function AskRunner({ topicId }: { topicId: number }) {
  const [prompt, setPrompt] = useState("");
  const [answer, setAnswer] = useState("");
  const [loading, setLoading] = useState(false);
  const [err, setErr] = useState<string | null>(null);

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setLoading(true);
    setAnswer("");
    setErr(null);
    try {
      const res = await fetch("/api/ai/ask", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ prompt, topicId }),
      });
      if (!res.ok) {
        const data = await res.json().catch(() => ({}));
        throw new Error(data.error ?? "失敗しました");
      }
      const reader = res.body?.getReader();
      const decoder = new TextDecoder();
      if (reader) {
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
          setAnswer((a) => a + decoder.decode(value, { stream: true }));
        }
      }
    } catch (e) {
      setErr((e as Error).message);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="space-y-4">
      <form onSubmit={handleSubmit} className="space-y-3">
        <textarea
          value={prompt}
          onChange={(e) => setPrompt(e.target.value)}
          placeholder="例: なぜ -3 の絶対値は 3 なの？"
          className="w-full border rounded-lg p-3 min-h-32"
          required
        />
        <button
          type="submit"
          disabled={loading || !prompt.trim()}
          className="w-full bg-emerald-600 text-white font-bold py-3 rounded-lg disabled:opacity-50"
        >
          {loading ? "考え中..." : "AIにきく"}
        </button>
      </form>
      {err && <p className="text-red-600 text-sm">{err}</p>}
      {answer && (
        <article className="bg-white rounded-xl p-4 border whitespace-pre-wrap">
          {answer}
        </article>
      )}
    </div>
  );
}
```

```tsx
"use client";
import { useEffect, useState } from "react";

type Question = {
  id: number;
  type: "multi_choice" | "short_answer" | "numeric";
  questionText: string;
  choices: string[] | null;
};

type Mode = "loading" | "locked" | "ready" | "running" | "done";

type Result = { score: number; correct: number; total: number };

export function MockExamRunner() {
  const [mode, setMode] = useState<Mode>("loading");
  const [examId, setExamId] = useState<number | null>(null);
  const [questions, setQuestions] = useState<Question[]>([]);
  const [answers, setAnswers] = useState<Record<string, string>>({});
  const [result, setResult] = useState<Result | null>(null);

  useEffect(() => {
    fetch("/api/mock-exam/availability")
      .then((r) => r.json())
      .then((data) => {
        setMode(data.unlocked ? "ready" : "locked");
      });
  }, []);

  async function start() {
    const res = await fetch("/api/mock-exam/start", { method: "POST" });
    const data = await res.json();
    setExamId(data.examId);
    setQuestions(data.questions);
    setMode("running");
  }

  async function submit() {
    if (!examId) return;
    const res = await fetch("/api/mock-exam/submit", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ examId, answers }),
    });
    setResult(await res.json());
    setMode("done");
  }

  if (mode === "loading") return <p>読み込み中...</p>;
  if (mode === "locked") {
    return (
      <p className="text-stone-700">
        テスト1週間前から解放されます。テスト日を設定していない場合は{" "}
        <a href="/profile" className="underline">プロフィール</a> から設定してください。
      </p>
    );
  }
  if (mode === "ready") {
    return (
      <button
        type="button"
        onClick={start}
        className="w-full bg-emerald-600 text-white font-bold py-3 rounded-lg"
      >
        模試をはじめる（5教科×5問）
      </button>
    );
  }
  if (mode === "done" && result) {
    return (
      <div className="space-y-2">
        <h2 className="text-2xl font-bold">スコア: {result.score}点</h2>
        <p>{result.correct} / {result.total} 問正解</p>
      </div>
    );
  }
  return (
    <div className="space-y-6">
      {questions.map((q, i) => (
        <div key={q.id} className="space-y-2 border-b pb-4">
          <p className="font-medium">{i + 1}. {q.questionText}</p>
          {q.type === "multi_choice" && q.choices ? (
            <div className="space-y-1">
              {q.choices.map((c) => (
                <label key={c} className="flex items-center gap-2">
                  <input
                    type="radio"
                    name={`q-${q.id}`}
                    value={c}
                    checked={answers[String(q.id)] === c}
                    onChange={(e) =>
                      setAnswers((a) => ({ ...a, [String(q.id)]: e.target.value }))
                    }
                  />
                  <span>{c}</span>
                </label>
              ))}
            </div>
          ) : (
            <input
              type="text"
              value={answers[String(q.id)] ?? ""}
              onChange={(e) =>
                setAnswers((a) => ({ ...a, [String(q.id)]: e.target.value }))
              }
              className="w-full border rounded-lg px-3 py-2"
            />
          )}
        </div>
      ))}
      <button
        type="button"
        onClick={submit}
        className="w-full bg-emerald-600 text-white font-bold py-3 rounded-lg"
      >
        提出する
      </button>
    </div>
  );
}
```

`public/sw.js`:

```js
const CACHE = "chu1-v1";
const SHELL = ["/", "/login", "/home", "/manifest.webmanifest", "/icon-192.png", "/icon-512.png"];

self.addEventListener("install", (event) => {
  event.waitUntil(caches.open(CACHE).then((c) => c.addAll(SHELL)));
  self.skipWaiting();
});

self.addEventListener("activate", (event) => {
  event.waitUntil(
    caches.keys().then((keys) =>
      Promise.all(keys.filter((k) => k !== CACHE).map((k) => caches.delete(k)))
    )
  );
  self.clients.claim();
});

self.addEventListener("fetch", (event) => {
  const url = new URL(event.request.url);
  if (url.pathname.startsWith("/api")) {
    return;
  }
  event.respondWith(
    caches.match(event.request).then((cached) =>
      cached || fetch(event.request).catch(() => caches.match("/"))
    )
  );
});
```
